program {
    // Programa para determinar si tres números están en orden creciente
    int x, y, z, result;

    x = 1;
    y = 2;
    z = 3;

    read x;
    read y;
    read z;

    if (x < y and y < z) then {
        result = 1; // Orden creciente
    } else {
        result = 0; // No está en orden creciente
    } fi

    write result;
}
